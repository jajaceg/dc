kolor(a, zolty) :- panstwo(a), paleta(zolty), not innyKolor(a, zolty).
kolor(b, czerwony) :- panstwo(b), paleta(czerwony), not innyKolor(b, czerwony).
kolor(b, niebieski) :- panstwo(b), paleta(niebieski), not innyKolor(b, niebieski).
kolor(b, zolty) :- panstwo(b), paleta(zolty), not innyKolor(b, zolty).

innyKolor(a, czerwony) :- panstwo(a), paleta(czerwony), not kolor(a, czerwony).
innyKolor(a, niebieski) :- panstwo(a), paleta(niebieski), not kolor(a, niebieski).
innyKolor(a, zolty) :- panstwo(a), paleta(zolty), not kolor(a, zolty).

maKolor(a) :- panstwo(a), paleta(niebieski), kolor(a, niebieski).
maKolor(a) :- panstwo(a), paleta(zolty), kolor(a, zolty).

maKolor(b) :- panstwo(b), paleta(niebieski), kolor(b, niebieski).

:- panstwo(b), not maKolor(b).

:- panstwo(b), paleta(niebieski), paleta(zolty), kolor(b, niebieski), kolor(b, zolty), niebieski != zolty.
